import datetime
import re
import socket
import time,json
import threading
import config

list_th = []
def convert_time():
    t = datetime.datetime.now().strftime('%Y%m%d%H%M%S')[2:]
    t_hex = ''
    for i in range(int(len(t)/2)):t_hex += str(hex(int(t[2*i:2*i+2])))[2:].rjust(2,'0').upper()
    return t_hex

def convert_vin(vin):
    vin_hex = ''
    for i in vin: vin_hex += str(hex(ord(i))[2:].rjust(2, '0')).upper()
    return vin_hex

def convert_iccid(iccid):
    iccid_hex = ''
    for i in iccid: iccid_hex += str(hex(ord(i))[2:].rjust(2, '0')).upper()
    return iccid_hex

def bcc(bcc_input):
    bcc = 0
    for i in range(int(len(bcc_input) / 2) - 1):
        bcc = bcc ^ int(bcc_input[2 * i:2 * (i + 1)], 16)
    return hex(bcc).format()[-2:].upper()

def heart(s,vin):
    while True:
        with open('qqq.json','r') as f:
            m = json.load(f)
        if m['mark'] == 'q':
            break
        else:
            msg_07 = config.msg_232307
            msg_07 = msg_07[0:8] + convert_vin(vin) + msg_07[42:]
            msg_07 = msg_07[:-2] + bcc(msg_07)
            s.sendall(bytes.fromhex(msg_07))
            with open(f'./log/232307_&#123;vin&#125;.txt', 'a') as f:
                f.write(msg_07)
                f.flush()
            time.sleep(10)

def login(s,vin,iccid):
    msg_01 = config.msg_232301
    msg_01 = msg_01[0:8] + convert_vin(vin) + msg_01[42:]
    msg_01 = msg_01[0:48] + convert_time() + msg_01[60:]
    msg_01 = msg_01[0:64] + convert_iccid(iccid) + msg_01[104:]
    msg_01 = msg_01[:-2] + bcc(msg_01)
    s.sendall(bytes.fromhex(msg_01))
    msg = ''
    while True:
        recv_data = s.recv(1024)
        if recv_data:
            msg += recv_data.hex()
            break
        else:
            continue
    if '23230101' in msg:
        r = '登入成功\n'
    else:
        r = '登入失败\n'
    with open(f'./log/&#123;vin&#125;.txt', 'a') as f:
        f.write(r)
        f.flush()

def quit():
    m = input('q结束：')
    with open('qqq.json','w') as f:
        json.dump(&#123;"mark":"q"&#125;,f)

def recv_data(s,vin):
    count = 1
    while True:
        with open('qqq.json','r') as f:
            m = json.load(f)
        if m['mark'] == 'q':
            print(2)
            break
        else:
            recv_data = s.recv(1024)
            msg_list = []
            if recv_data:
                msg = recv_data.hex()
                if msg.count('2323') > 1:
                    index_list = [i.start() for i in re.finditer('2323', msg)]
                    for i in range(len(index_list)-1):
                        msg_list.append(msg[index_list[i]:index_list[i+1]])
                    msg_list.append(msg[index_list[-1]:])
                else:
                    msg_list.append(msg)
                for msg in msg_list:
                    if '232380' in msg or '232381' in msg or '232382' in msg:
                        with open(f'./log/&#123;vin&#125;.txt','a') as f:
                            f.write(f'*****第&#123;count&#125;次*****' + '\n'+
                                str(datetime.datetime.now()) + ' receive: ' + msg+'\n')
                            f.flush()
                        count += 1
                        if '232380' in msg:
                            str(hex(int(msg[44:48],16) + 1))[2:].rjust(2,'0').upper()
                            send_msg = msg[0:6] + '01' + msg[8:44] + str(hex(int(msg[44:48],16) + 1))[2:].rjust(4,'0').upper() + msg[48:64] + '01' + msg[-2:]
                            send_msg = send_msg[0:48] + convert_time() + send_msg[60:]
                            send_msg = send_msg[:-2] + bcc(send_msg)
                            s.sendall(bytes.fromhex(send_msg))
                        else:
                            send_msg = msg[0:6] + '01' + msg[8:]
                            send_msg = send_msg[0:8] + convert_vin(vin) + send_msg[42:]
                            send_msg = send_msg[0:48] + convert_time() + send_msg[60:]
                            send_msg = send_msg[:-2] + bcc(send_msg)
                            s.sendall(bytes.fromhex(send_msg))
                        with open(f'./log/&#123;vin&#125;.txt','a') as f:
                            f.write(str(datetime.datetime.now()) + ' send: ' + send_msg+'\n\n')
                            f.flush()
                    elif '232307' in msg:
                        with open(f'./log/232307_&#123;vin&#125;.txt','a') as f:
                            f.write(str(datetime.datetime.now()) + ' receive: ' + msg+'\n\n')
                            f.flush()
            continue
